#[derive(FromQueryResult)]
struct ArticleView {
    id: i32,
    title: String,
    author_name: Option<String>,
}

Query:

let data = Article::find()
    .select_only()
    .column(article::Column::Id)
    .column(article::Column::Title)
    .column_as(author::Column::Name, "author_name")
    .join(JoinType::LeftJoin, article::Relation::Author.def())
    .into_model::<ArticleView>()
    .all(db)
    .await?;
